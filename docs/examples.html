PID Controller
from controltheorylib import System, PID, Visualizer

plant = System(transfer_function="1/(s^2 + s)")
controller = PID(kp=1, ki=0.5, kd=0.1)
closed_loop = controller * plant

viz = Visualizer()
viz.plot_step_response(closed_loop)
viz.render()
View Animation
State Space System
from controltheorylib import System, Visualizer
import numpy as np

A = np.array([[0, 1], [-1, -0.5]])
B = np.array([[0], [1]])
C = np.array([[1, 0]])
D = np.array([[0]])

sys = System(state_space=(A, B, C, D))
viz = Visualizer()
viz.plot_step_response(sys)
viz.render()
